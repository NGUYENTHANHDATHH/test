            // â‘£ Create input stream to read client message

        } catch (Exception e) {
            e.printStackTrace();

            try {
                if (reader != null) reader.close();
                if (socket != null) socket.close();
                if (serverSocket != null) serverSocket.close();
            } catch (IOException e) { /* ignore */ }
        }
    }
